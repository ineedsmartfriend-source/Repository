<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>One-Click OCR Memo</title>
<script src="https://cdn.jsdelivr.net/npm/@tailwindcss/browser@4"></script>
<script src="https://cdn.jsdelivr.net/npm/tesseract.js@5/dist/tesseract.min.js"></script>
</head>
<body class="bg-gray-900 text-white p-4">

<div class="fixed bottom-5 right-5 z-50">
<button id="capture-btn" class="bg-blue-600 hover:bg-blue-700 text-white font-bold p-4 rounded-full shadow-lg transition-all transform hover:scale-110">
📸 텍스트 추출
</button>
</div>

<div class="max-w-md mx-auto mt-10">
<h1 class="text-xl font-bold mb-4">📝 저장된 메모</h1>
<div id="memo-list" class="space-y-2">
</div>
</div>

<script>
const captureBtn = document.getElementById('capture-btn');
const memoList = document.getElementById('memo-list');

// 로컬스토리지에서 기존 메모 불러오기
let mamos = JSON.parse(localStorage.getItem('my_memos')) || [];
renderMemos();

captureBtn.addEventListener('click', async () => {
try {
// 1. 스마트폰 화면 캡처 스트림 가져오기
const stream = await navigator.mediaDevices.getDisplayMedia({ video: true });
const video = document.createElement('video');
video.srcObject = stream;
await video.play();

// 2. 캔버스에 현재 프레임 그리기
const canvas = document.createElement('canvas');
canvas.width = video.videoWidth;
canvas.height = video.videoHeight;
canvas.getContext('2d').drawImage(video, 0, 0);

// 스트림 종료
stream.getTracks().forEach(track => track.stop());

// 3. Tesseract.js로 한글 OCR 수행
captureBtn.innerText = "⏳ 추출 중...";
const { data: { text } } = await Tesseract.recognize(canvas, 'kor');

// 4. 메모 저장
if(text.trim()) {
mamos.push({ id: Date.now(), content: text });
localStorage.setItem('my_memos', JSON.stringify(mamos));
renderMemos();
alert('메모가 저장되었습니다!');
} else {
alert('추출된 텍스트가 없습니다.');
}
captureBtn.innerText = "📸 텍스트 추출";

} catch (err) {
console.error("오류 발생:", err);
captureBtn.innerText = "📸 텍스트 추출";
}
});

function renderMemos() {
memoList.innerHTML = mamos.map(m => &lt;div class="bg-gray-800 p-3 rounded-lg border border-gray-700"&gt; &lt;p class="text-sm"&gt;${m.content}&lt;/p&gt; &lt;/div&gt;).join('');
}
</script>
